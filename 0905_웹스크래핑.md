## 검색어를 입력받아 url 열기
```py
# 검색어를 입력받아 url 열기

# webbrowser import하기
import webbrowser
import time
# 검색어 입력받기
query = input("검색어를 입력하세요 : ")
naver = "https://search.naver.com/search.naver?where=nexearch&sm=top_hty&fbm=1&ie=utf8&query="
# naver 뒤에 검색어 이어붙이기
url = naver+query

# url페이지 오픈
for i in range(4):
    webbrowser.open(url)
    time.sleep(2)
```

## 웹페이지 html파일 가져오기

# requests import하기
import requests
